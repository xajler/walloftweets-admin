Authenticate

For given email and password authenticates user aginst MongoDB user collection. Uses auth helper utility to hash the password and salt to validate user.
Returns a user as a callback.

authenticate = (email, password, callback) ->
  User.findOne 
    'email': email
  , (err, user) ->
    if err then console.log err else console.log user
    unless user
      console.log email + ' ' + password
      callback null

    auth = new Auth()

    if auth.validate(password, user.salt, user.password)
      callback user
      return
    callback null

Save

Saves the given user to the MongoDB user collection.

save = (user) ->
  user.save (err) ->
    if err then throw err else console.log 'User saved!'

createDefaultUser = ->
